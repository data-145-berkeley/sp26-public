Let $X_1, X_2, \ldots, X_n$ be i.i.d.\ random variables with mean $\mu$ and
variance $\sigma^2$. Let $\bar{X} = \frac{1}{n}\sum_{i=1}^n X_i$.

\begin{enumerate}[label=(\alph*)]
    \item Show that $\E[\bar{X}] = \mu$.

    

    \item Show that $\Var(\bar{X}) = \sigma^2/n$.

    
\end{enumerate}
